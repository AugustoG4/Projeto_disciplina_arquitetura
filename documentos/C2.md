C4Container
title C2 - Diagrama de Containers: Sistema de Gestão da Clínica

Person(paciente, "Paciente")
Person(profissional, "Psicólogo")
Person(secretaria, "Secretária")
Person(administrador, "Administrador")

System_Boundary(sistema, "Sistema de Gestão da Clínica") {

    Container(webapp, "Front-end Web", "React / Vue / HTML5 + Tailwind", 
        "Interface acessada por pacientes, psicólogos, secretárias e administradores.")
    
    Container(api, "Back-end API", "Node.js / Django / Flask", 
        "Processa lógica de negócio, autenticação, agendamentos, prontuário e financeiro.")
    
    ContainerDb(db, "Banco de Dados", "PostgreSQL", 
        "Armazena pacientes, agenda, pagamentos, evoluções clínicas, sessões, usuários.")

    Container(mobile, "Acesso Mobile (PWA)", "PWA integrada ao front-end",
        "Versão mobile responsiva para uso em celulares por profissionais e pacientes.")

}

System_Ext(whatsapp, "API WhatsApp", "Envio de notificações e lembretes.")
System_Ext(googleAgenda, "Google Calendar API")
System_Ext(pgto, "Gateway de Pagamento (Pix/Cartão)")

Rel(paciente, webapp, "Usa")
Rel(profissional, webapp, "Usa")
Rel(secretaria, webapp, "Usa")
Rel(administrador, webapp, "Usa")

Rel(webapp, api, "Requisições HTTPS (REST/JSON)")
Rel(api, db, "CRUD completo", "SQL")

Rel(api, whatsapp, "Envio de lembretes")
Rel(api, googleAgenda, "Sincroniza agenda")
Rel(api, pgto, "Solicita transações / recebe callbacks")
